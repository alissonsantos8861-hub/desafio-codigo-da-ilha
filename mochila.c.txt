#include <stdio.h>
#include <string.h>
#include <stdbool.h>

#define MAX_ITENS 10

// =========================
//       ESTRUTURA
// =========================
typedef struct {
    char nome[50];
    char tipo[50];
    int quantidade;
    int prioridade; // Usado apenas no n칤vel Mestre
} Item;

// Enum para ordena칞칚o (n칤vel Mestre)
typedef enum {
    ORD_NOME = 1,
    ORD_TIPO = 2,
    ORD_PRIORIDADE = 3
} Criterio;

// =========================
//   FUN칂칏ES AUXILIARES
// =========================

// Listar itens
void listarItens(Item mochila[], int total) {
    if (total == 0) {
        printf("\nA mochila est치 vazia!\n");
        return;
    }

    printf("\n=== ITENS NA MOCHILA ===\n");
    printf("%-20s %-15s %-10s %-10s\n", "Nome", "Tipo", "Qtd", "Prioridade");
    printf("-------------------------------------------------------\n");

    for (int i = 0; i < total; i++) {
        printf("%-20s %-15s %-10d %-10d\n",
               mochila[i].nome, mochila[i].tipo,
               mochila[i].quantidade, mochila[i].prioridade);
    }
}

// Adicionar item
void adicionarItem(Item mochila[], int *total) {
    if (*total >= MAX_ITENS) {
        printf("\nA mochila est치 cheia!\n");
        return;
    }

    Item novo;

    printf("\nNome do item: ");
    scanf(" %[^\n]", novo.nome);

    printf("Tipo do item: ");
    scanf(" %[^\n]", novo.tipo);

    printf("Quantidade: ");
    scanf("%d", &novo.quantidade);

    printf("Prioridade (1 a 5): ");
    scanf("%d", &novo.prioridade);

    mochila[*total] = novo;
    (*total)++;

    printf("\nItem adicionado com sucesso!\n");
}

// Remover item
void removerItem(Item mochila[], int *total) {
    if (*total == 0) {
        printf("\nA mochila est치 vazia!\n");
        return;
    }

    char nomeBusca[50];
    printf("\nNome do item a remover: ");
    scanf(" %[^\n]", nomeBusca);

    for (int i = 0; i < *total; i++) {
        if (strcmp(mochila[i].nome, nomeBusca) == 0) {
            for (int j = i; j < *total - 1; j++) {
                mochila[j] = mochila[j + 1];
            }
            (*total)--;
            printf("\nItem removido!\n");
            return;
        }
    }

    printf("\nItem n칚o encontrado!\n");
}

// Busca Sequencial (n칤vel Aventureiro)
void buscarSequencial(Item mochila[], int total) {
    char nomeBusca[50];
    printf("\nNome do item a buscar: ");
    scanf(" %[^\n]", nomeBusca);

    for (int i = 0; i < total; i++) {
        if (strcmp(mochila[i].nome, nomeBusca) == 0) {
            printf("\n=== Item Encontrado ===\n");
            printf("Nome: %s\nTipo: %s\nQuantidade: %d\nPrioridade: %d\n",
                mochila[i].nome, mochila[i].tipo, mochila[i].quantidade, mochila[i].prioridade);
            return;
        }
    }

    printf("\nItem n칚o encontrado!\n");
}

// Ordena칞칚o Insertion Sort (n칤vel Mestre)
int ordenar(Item mochila[], int total, Criterio crit) {
    int comparacoes = 0;

    for (int i = 1; i < total; i++) {
        Item chave = mochila[i];
        int j = i - 1;

        if (crit == ORD_NOME) {
            while (j >= 0 && strcmp(mochila[j].nome, chave.nome) > 0) {
                mochila[j + 1] = mochila[j];
                j--;
                comparacoes++;
            }
        } else if (crit == ORD_TIPO) {
            while (j >= 0 && strcmp(mochila[j].tipo, chave.tipo) > 0) {
                mochila[j + 1] = mochila[j];
                j--;
                comparacoes++;
            }
        } else if (crit == ORD_PRIORIDADE) {
            while (j >= 0 && mochila[j].prioridade > chave.prioridade) {
                mochila[j + 1] = mochila[j];
                j--;
                comparacoes++;
            }
        }

        mochila[j + 1] = chave;
    }

    return comparacoes;
}

// Busca Bin치ria (n칤vel Mestre)
void buscaBinaria(Item mochila[], int total) {
    char nomeBusca[50];
    printf("\nNome do item a buscar (bin치rio): ");
    scanf(" %[^\n]", nomeBusca);

    int inicio = 0, fim = total - 1;

    while (inicio <= fim) {
        int meio = (inicio + fim) / 2;
        int cmp = strcmp(nomeBusca, mochila[meio].nome);

        if (cmp == 0) {
            printf("\n=== Item Encontrado ===\n");
            printf("Nome: %s\nTipo: %s\nQuantidade: %d\nPrioridade: %d\n",
                   mochila[meio].nome, mochila[meio].tipo,
                   mochila[meio].quantidade, mochila[meio].prioridade);
            return;
        }

        if (cmp > 0)
            inicio = meio + 1;
        else
            fim = meio - 1;
    }

    printf("\nItem n칚o encontrado!\n");
}

int main() {
    Item mochila[MAX_ITENS];
    int total = 0;
    int opcao, criterio;
    bool ordenadoPorNome = false;

    do {
        printf("\n=== DESAFIO: C칍DIGO DA ILHA ===\n");
        printf("1 - Adicionar item\n");

        printf("Escolha: ");
        scanf("%d", &opcao);

        switch (opcao) {
            case 1:
                adicionarItem(mochila, &total);
                ordenadoPorNome = false;
                break;
            case 2:
                removerItem(mochila, &total);
                ordenadoPorNome = false;
                break;
            case 3:
                listarItens(mochila, total);

            case 5:
                printf("\nOrdenar por:\n1 - Nome\n2 - Tipo\n3 - Prioridade\nEscolha: ");

                printf("\nItens ordenados! Compara칞칫es: %d\n", comp);

                ordenadoPorNome = (criterio == ORD_NOME);
                break;
            case 6:
                if (!ordenadoPorNome) {
                    printf("\nERRO: A lista precisa estar ordenada por NOME!\n");
                } else {
                    buscaBinaria(mochila, total);
                }
                break;
            case 0:
                printf("\nSaindo... Boa sorte na ilha! 游댦游끻勇끂n");
                break;
            default:
                printf("\nOp칞칚o inv치lida!\n");
        }

    } while (opcao != 0);

}
